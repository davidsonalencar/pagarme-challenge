(function () {
    'use strict';

    angular.module('recipients').controller('RecipientsFormController', [
        'Recipients',
        '$q',
        '$state',
        '$stateParams',
        '$filter',
        RecipientsFormController
    ]);

    function RecipientsFormController(Recipients, $q, $state, $stateParams, $filter) {
        
        //-- private variables
        var ctrl = this;
        
        //-- public variables
        ctrl.recipient = new Recipients({
            transfer_interval: ctrl.transferIntervalOptions[0].id,
            transfer_enabled: true,
            legal_name: ''
        });
        
        ctrl.transferIntervalOptions = $.map(['daily', 'weekly', 'monthly'], function(value){
            return {
                id: value,
                name: $filter('recipientsTransferIntervalLabel')(value)
            };
        });
        
        //-- private methods

        //-- public methods
        ctrl.save = save;

        return init();
        
        //-- initialization methods

        /**
         * Realiza a inicialização do controller, configuração de eventos e 
         * definição de watcher(s)
         * @returns {void}
         */
        function init() {
            
            changeTitleForm();
            
            if (angular.isDefined($stateParams.id)) {
                getRecipient($stateParams.id).then(function(recipient){
                    ctrl.recipient = recipient;
                });
            } 
                        
        }
        
        function save(recipient) {
            
            var promise;
            if (angular.isDefined($stateParams.id)) {
                promise = recipient.$update();
            } else {
                promise = recipient.$save();
            }
            promise.then(function(){
                $state.go('^');
            });
            
        }
        
        function getRecipient(id) {
            
            return $q(function(resolve, reject){
                Recipients.get({
                    id: id
                }, resolve, reject);
            });
            
        }
        
        function changeTitleForm() {
            
            if (angular.isDefined($stateParams.id)) {
                ctrl.titleForm = 'Alterar';
            } else {
                ctrl.titleForm = 'Adicionar';
            }
            
        }
        
    }
    
})();
